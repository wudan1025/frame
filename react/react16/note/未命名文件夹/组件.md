function Clock(props) {
  return (
    <div>
      <h1>hello world</h1>
      <h2>现在是 {props.data.toLocaleTimeString()}</h2>
    </div>
  );
}

function tick2() {
  ReactDOM.render(
    <Clock data={new Date()} />,
    document.getElementById('example')
  );
}

class Clock3 extends React.Component {
  render() {
    return (
      <div>
        <h1>hello</h1>
        <h2>3 现在是 {this.props.data.toLocaleTimeString()}</h2>
      </div>
    );
  }
}

function tick3() {
  ReactDOM.render(
    <Clock3 data={new Date()}></Clock3>,

  );
}
